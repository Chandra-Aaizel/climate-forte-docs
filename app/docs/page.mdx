export const getBadgeColors = (title) => {
  switch (title) {
    case "Air quality":
      return "bg-green-400"
    case "Fire":
      return "bg-red-400"
    case "Core":
      return "bg-blue-600"
    case "Maritime":
      return "bg-teal-400"
    case "Flood":
      return "bg-blue-200"
    case "Lightning":
      return "bg-gray-400"
    case "Soil":
      return "bg-yellow-300"
    case "Solar":
      return "bg-purple-400"
    case "Pollen":
      return "bg-indigo-900"
    default:
      return "bg-gray-200"
  }
}
